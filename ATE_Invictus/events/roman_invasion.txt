	id = RIATEI.1
	hide_window = yes

	has_dlc = "Sunset Invasion"
	
	trigger = {
		OR = {
			AND = {
				has_game_rule = {
					name = roman_invasion
					value = historical
				}
				year = 2667
			}
			AND = {
				has_game_rule = {
					name = roman_invasion
					value = random
				}
				total_years_played = 30
			}
			AND = {
				has_game_rule = {
					name = roman_invasion
					value = delayed_random
				}
				year = 2750
				total_years_played = 50
			}
		}
		NOR = {
			year = 2860
			has_global_flag = roman_explorers 
			has_game_rule = {
				name = roman_invasion
				value = off
			}
		}
		OR = {
			province_id = 1124 # New Orleans
			province_id = 1263 # Galveston
			province_id = 1251 # Atchafalaya
		}
	}
	
	immediate = {
		if = { 
			limit = {  
				OR = {
					has_game_rule = {
						name = roman_invasion
						value = random
					}
					has_game_rule = {
						name = roman_invasion
						value = delayed_random
					}
				}
				NOT = { year = 2860 } #Will never fire otherwise.
			}
			random = { 
				chance = 40
					modifier = {
						factor = 0.75
						total_years_played = 100
					}
					modifier = {
						factor = 0.5
						total_years_played = 150
					}
					modifier = {
						factor = 0.25
						total_years_played = 200
					}
				break = yes #Chance of misfire to prevent the randomized invasions from firing immediately after 50 years of game.
			}
		}
		log = "roman INVASION IMMEDIATE 5532432005"
		set_global_flag = roman_explorers
		set_province_flag = roman_explorers
		activate_title = { title = e_roman_empire_outside status = yes }
		owner = { character_event = { id = RIATEI.74 } }
		
		k_papal_state = {
			holder_scope = {
				create_random_soldier = {
					random_traits = no
					culture = Roman
					religion = hellenic_pagan
					age = 20
					attributes = {
						martial = 8
						diplomacy = 6
						stewardship = 6
						intrigue = 6
						learning = 6
					}
					health = 6
					fertility = 0.7
					trait = brilliant_strategist
					trait = genius
					trait = strong
					trait = zealous
					trait = ambitious
					trait = diligent
					trait = immortal
					trait = patient
					flag = horde_invader
				}
				new_character = {
					e_roman_empire_outside = {
						grant_title = PREV
					}
					set_government_type = feudal_government
					e_roman_empire_outside = {
						add_law = succ_primogeniture
						add_law = agnatic_succession
						if = {
							limit = {
								has_dlc = "Zeus"
							}
							add_law = revoke_title_law_2
						}
						if = {
							limit = {
								NOT = { has_dlc = "Zeus" }
							}
							add_law = crown_authority_1
						}
					}
					set_defacto_liege = THIS
					narrative_event = { id = RIATEI.2 days = 7 }
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 720
		modifier = {
			factor = 0.5
			year = 2830
		}
		modifier = {
			factor = 4
			has_game_rule = {
				name = roman_invasion
				value = random
			}
		}
		modifier = {
			factor = 3
			has_game_rule = {
				name = roman_invasion
				value = delayed_random
			}
		}
	}
	
	option = {
		name = EVTOPTA_RIATEI_001
	}
}
